Context: The current page context. */ -}} {{- $breadcrumb_show := $.Param "breadcrumb" -}} {{- $url := replace .Permalink $.Site.Home.Permalink "" -}} {{- $url = trim $url "/" -}} {{- if and $breadcrumb_show $url -}}
    {{- with $.Site.Home -}}
  1. {{- $target_url := .RelPermalink -}} {{- $allow_url_attrs := false -}} {{- with .Params.target -}} {{- $target_url = . -}} {{- $allow_url_attrs = true -}} {{- end -}} {{- partial "render-link" (dict "Page" $.Page "Destination" $target_url "Title" .LinkTitle "InnerHTML" "" "AllowUrlAttrs" $allow_url_attrs) -}}
    2. {{- end -}} {{- $breadcrumb_depth := $.Param "breadcrumb_depth" -}} {{- if not $breadcrumb_depth -}} {{- $breadcrumb_depth = 3 -}} {{- end }} {{- $breadcrumb_depth = $breadcrumb_depth | add -1 -}} {{- $breadcrumb_pages := slice -}} {{- $breadcrumb_url := "" -}} {{- range split $url "/" -}} {{- if . -}} {{- $breadcrumb_url = print $breadcrumb_url "/" . -}} {{- with $.Site.GetPage $breadcrumb_url }} {{ if ne . $.Page }} {{- $breadcrumb_pages = $breadcrumb_pages | append . -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- $breadcrumb_count := len $breadcrumb_pages -}} {{- if lt $breadcrumb_count $breadcrumb_depth -}} {{- $breadcrumb_depth = $breadcrumb_count -}} {{- else if gt $breadcrumb_count $breadcrumb_depth -}}
  2. {{ T "ellipsis" }}
    3. {{- end -}} {{- range $breadcrumb_pages | after (sub (len $breadcrumb_pages) $breadcrumb_depth) -}}
  3. {{- $target_url := .RelPermalink -}} {{- $allow_url_attrs := false -}} {{- with .Params.target -}} {{- $target_url = . -}} {{- $allow_url_attrs = true -}} {{- end -}} {{- partial "render-link" (dict "Page" $.Page "Destination" $target_url "Title" "" "InnerHTML" .LinkTitle "AllowUrlAttrs" $allow_url_attrs) -}}
    4. {{- end -}}
  4. {{ .LinkTitle }}
{{- end -}}
